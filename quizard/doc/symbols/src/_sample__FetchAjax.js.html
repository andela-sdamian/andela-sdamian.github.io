  7 /**
  8  * @constructor
  9  * Fetch Questions from any source synchronously not asynchronous this is to ensure the quiz is loaded before taking the test
 10  * @param {string} url - The source url where questions are kept
 11  * @param {object} callback - A callback function That executes after the question is loaded
 12  * */
 13 function FetchAjax(url, callback){
 14 
 15     this.data = "";
 16     this.request = new XMLHttpRequest();
 17 
 18     this.request.open('GET', url, false);
 19 
 20     var temReq = this.request;
 21 
 22     /**
 23      * Listens to the ready state change event
 24      * @private
 25      * */
 26     function reqReadyStateChange(){
 27 
 28         if(temReq.readyState == 4){
 29             if(temReq.status == 200){
 30 
 31                 callback(temReq.responseText);
 32             }else{
 33 
 34                 return "Error occurred";
 35             }
 36         }
 37     }
 38 
 39     this.request.onreadystatechange = reqReadyStateChange;
 40 }
 41 
 42 /**
 43  * sends the ajax request to the server
 44  * */
 45 FetchAjax.prototype.send = function(){
 46 
 47     this.request.send(null);
 48 };
