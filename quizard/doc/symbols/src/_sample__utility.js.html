 10 function handleSelectedOption(){
 11     var form = document.frmOptions;
 12     var options = form.option;
 13     for(var index = 0, max = options.length; index < max; index++){
 14 
 15         if(options[index].checked){
 16             return options[index].value;
 17         }
 18 
 19     }
 20     return "";
 21 }

 32 /**
 33  * This function generateOption is responsible for dynamically generating options and placing them on the DOM
 34  * @param {string} name - The name of the radio button
 35  * @param {string} value - The value in the radio button
 36  * @param {string} optionText - The text beside the radio button
 37  * @param {string} [klass = 'form-control'] - Optional class parameter
 38  *
 39  * */
 40 
 41 function generateOption(name, value, optionText, klass){
 42 
 43     var radio = create("input"),
 44         label = create("label"),
 45         textNode = document.createTextNode(optionText);
 46 
 47     label.appendChild(textNode);
 48     label.insertBefore(radio,textNode);
 49     label.setAttribute("class", klass || "form-control");
 50 
 51     radio.type = "radio";
 52     radio.name = name;
 53     radio.value = value;
 54 
 55     return label;
 56 }
